\node[dag_lemma, text width=4.6cm] (Lemmas78) at (17.7,-8.2) {

\node[dag_lemma, text width=4.6cm] (Lemmas910) at (17.7,-12.1) {

\node[dag_result, text width=4.8cm] (EndpointFeasible) at (23.6,-12.1) {

\node[dag_conditional, text width=4.6cm] (Thm4) at (23.6,-8.2) {
  \textbf{Theorem 4 (App. D.4)}\\

};

\node[dag_conditional, text width=4.6cm] (Thm2) at (29.5,-8.2) {

\node[dag_conditional, text width=4.6cm] (Thm3) at (23.6,-16.0) {
  \textbf{Theorem 3 (Sec. 4.2)}\\

};

\node[dag_result, text width=5.0cm] (MeasShapeSource) at (29.5,-16.0) {
