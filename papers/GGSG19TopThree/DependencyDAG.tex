  \textbf{Proposition 3} \\
  K-approval pairwise learning has the closed approval rate, or errors eventually vanish at the boundary
};

\node[dag_result] (Theorem1) at (0,-7.0) {
  \textbf{Theorem 1} \\
  a static convex combination of scoring rules is consistent and weakly dominates randomized scoring

\node[dag_result] (Prop4) at (7.0,-7.0) {
  \textbf{Proposition 4} \\

\node[dag_result] (Theorem2) at (14.0,-7.0) {
  \textbf{Theorem 2} \\
  some static K-approval cutoff weakly dominates every randomized K-approval pairwise rate
};

\node[dag_lemma] (MallowsSupport) at (21.0,-7.0) {
  \textbf{Mallows Support} \\
  boundary-pair pivotality and one-loser approval probability facts
};

\node[dag_result] (RandHelps) at (0,-10.5) {
  \textbf{Randomization-Helps Theorem} \\
  a constructed W-selection law has randomized approval beating every static K cutoff
};

\node[dag_result] (MallowsNoRandom) at (14.0,-10.5) {
  \textbf{Mallows Corollary} \\
  for Mallows $q<1$, an optimal static K-approval rule weakly dominates randomized K-approval
};

\node[dag_result] (MallowsNotWK) at (21.0,-10.5) {
  \textbf{Mallows W-Approval Theorem} \\
  a four-candidate high-noise Mallows instance shows W-approval is not approval-rate optimal
};
